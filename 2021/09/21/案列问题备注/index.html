  8. java 非法字符 \ufeff

    • 提示：

      Error:(1, 1) java: 非法字符: ‘\ufeff’
      Error:(1, 10) java: 需要class, interface或enum

      百度了下说是编码问题

    • 解决方法：

      1. 先选择gbk、然后convert、然后在选择一下utf-8在convert确认下就行了
      2. 将这些报错的文件拷贝到一个目录中，然后用notpad++打开，点击工具类上的 编码，将 以uft8格式编码 转为 以utf8无BOM格式编码，然后保存，再覆盖idea中对应的文件。

11. route_list.html 页面

  1. <script>
    $(function (){
        //设置或者返回从问号(?)开始的URL(查询部分)
        const search = location.search;  //?cid=5
        //按照参数字符将字符串分成数组，不包含参数字符：["?cid", "5"]
        const cid = search.split("=")[1];
        //当页码加载完成后，调用load方法，发送ajax请求加载数据
        load(cid);
        $("#jump").click(function (){
            const inputVal = $("#inputVal").val();
            load(cid, inputVal);
        });
    });
    //为了不是点击页码重新加载网页，需要将异步请求提出为一个函数，点击页码时执行这个函数进行异步请求
    function load(cid, currentPage) {
        //发送ajax请求，请求route/pageQuery,传递cid
        $.get("route/pageQuery", {cid : cid, currentPage : currentPage}, function (pb) {
            //解析pageBean数据，展示到页面上

            //1分页工具条数据展示
            //1.1展示总页码和总记录数
            $("#totalPage").html(pb.totalPage);
            $("#totalCount").html(pb.totalCount);
            //1.2展示分页页码
            /* <li><a href="">首页</a></li>
                 <li className="threeword"><a href="#">上一页</a></li>
                 <li><a href="#">1</a></li>
                 <li><a href="#">2</a></li>
                 <li><a href="#">3</a></li>
                 <li><a href="#">4</a></li>
                 <li><a href="#">5</a></li>
                 <li><a href="#">6</a></li>
                 <li><a href="#">7</a></li>
                 <li><a href="#">8</a></li>
                 <li><a href="#">9</a></li>
                 <li><a href="#">10</a></li>
                 <li className="threeword"><a href="javascript:;">下一页</a></li>
                 <li className="threeword"><a href="javascript:;">末页</a></li>*/

            let lis = "";
            const firstPage = '<li onclick="load(' + cid + ')"><a href="javascript:void(0);">首页</a></li>';
            //计算上一页的页码
            let beforeNum = pb.currentPage - 1;
            if (beforeNum === 0) {
                beforeNum = 1;
            }
            const beforePage = '<li onclick="load(' + cid + ', ' + beforeNum + ')" className="threeword" ><a href="javascript:void(0);" style="color: crimson"><<</a></li>';
            lis += firstPage;
            lis += beforePage;
            //1.2 展示分页页码
            /*
                    1.一共展示10个页码，能够达到前5后4的效果
                    2.如果前边不够5个，后边补齐10个
                    3.如果后边不足4个，前边补齐10个
                */
            //定义开是位置begin，结束位置end，
            let begin; //开始位置
            let end; //结束位置
            if (pb.totalPage <= 10) {
                //总页码不够10页
                begin = 1;
                end = pb.totalPage;
            } else {
                //总页码超过10页
                begin = pb.currentPage - 5;
                end = pb.currentPage + 4;
                //2.如果前边不够5个，后边补齐10个
                if (begin <= 0) {
                    begin = 1;
                    end = begin + 9;
                }
                //3.如果后边不足4个，前边补齐10个
                if (end >= pb.totalPage) {
                    end = pb.totalPage;
                    begin = pb.totalPage - 9;
                }
            }
            for (let i = begin; i <= end; i++) {
                let li ;
                //判断当前页码是否等于i
                if (i === pb.currentPage) {
                    li = '<li onclick="load(' + cid + ', ' + i + ')" class="curPage"><a href="javascript:void(0);">' + i + '</a></li>';
                } else {
                    li = '<li onclick="load(' + cid + ', ' + i + ')"><a href="javascript:void(0);">' + i + '</a></li>';
                }
                lis += li;
            }
            /*for (let i = 1; i <= pb.totalPage; i++) {
                    let li ;
                    //判断当前页码是否等于i
                    if (i === pb.currentPage) {
                        li = '<li onclick="load(' + cid + ', ' + i + ')" class="curPage"><a href="javascript:void(0);">' + i + '</a></li>';
                    } else {
                        li = '<li onclick="load(' + cid + ', ' + i + ')"><a href="javascript:void(0);">' + i + '</a></li>';
                    }
                    lis += li;
                }*/
            const lastPage = '<li onclick="load(' + cid + ', ' + pb.totalPage + ')" className="threeword"><a href="javascript:void(0);">末页</a></li>';
            let nextNum = pb.currentPage + 1;
            if (nextNum > pb.totalPage ) {
                nextNum = pb.currentPage;
            }
            const nextPage = '<li onclick="load(' + cid + ', ' + nextNum + ')" className="threeword" ><a href="javascript:void(0);" style="color: crimson">>></a></li>';
            lis += nextPage;
            lis += lastPage;
            $("#pageNum").html(lis);
            //2列表数据展示
            /*<li>
                    <div className="img"><img src="images/04-search_03.jpg" alt=""></div>
                    <div className="text1">
                        <p>【减100元 含除夕/春节出发】广州增城三英温泉度假酒店/自由行套票</p>
                        <br/>
                        <p>1-2月出发，网付立享￥1099/2人起！爆款位置有限，抢完即止！</p>
    </div>
                    <div className="price">
                        <p className="price_num">
                            <span>&yen;</span>
                            <span>299</span>
                            <span>起</span>
    </p>
                        <p><a href="route_detail.html">查看详情</a></p>
    </div>
    </li>*/
            let route_lis = "";
            for (let i = 0; i < pb.list.length; i++) {
                let route = pb.list[i];
                let li = '<li>\n' +
                    '                            <div class="img"><img src="' + route.rimage + '" style="width: 299px;"></div>\n' +
                    '                            <div class="text1">\n' +
                    '                                <p>' + route.rname + '</p>\n' +
                    '                                <br/>\n' +
                    '                                <p>' + route.routeIntroduce + '</p>\n' +
                    '                            </div>\n' +
                    '                            <div class="price">\n' +
                    '                                <p class="price_num">\n' +
                    '                                    <span>&yen;</span>\n' +
                    '                                    <span>' + route.price + '</span>\n' +
                    '                                    <span>起</span>\n' +
                    '                                </p>\n' +
                    '                                <p><a href="route_detail.html">查看详情</a></p>\n' +
                    '                            </div>\n' +
                    '                        </li>';
                route_lis += li;
            }
            $("#route").html(route_lis);
            //加载页面后跳转至页面上部
            scrollTo(0, 0);
        });
    }
</script>
